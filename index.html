Smart Controllers

Advanced search with validation & caching

impl Controller<User> for UserController {
    async fn search(
        &self,
        container: Arc<Container>,
        query: SearchParams,
    ) -> HttpResult<ElifResponse> {
        let pool = self.get_pool(&container).await?;
        
        let search_term = query.q.unwrap_or_default();
        if search_term.len() < 3 {
            return Err(HttpError::bad_request(
                "Search term must be at least 3 characters"
            ));
        }

        let users = User::query()
            .where_clause("name ILIKE $1 OR email ILIKE $1")
            .limit(50)
            .execute_with_params(&pool, vec![
                format!("%{}%", search_term)
            ])
            .await?;

        Ok(ElifResponse::json(ApiResponse::success(users)))
    }
}

Type-Safe ORM

Models with validation & timestamps

#[derive(Debug, Serialize, Deserialize, Clone)]
pub struct User {
    pub id: Uuid,
    pub name: String,
    pub email: String,
    pub age: Option<i32>,
    pub is_active: bool,
    pub created_at: DateTime<Utc>,
    pub updated_at: DateTime<Utc>,
}

impl Model for User {
    type PrimaryKey = Uuid;

    fn table_name() -> &'static str { "users" }
    fn uses_timestamps() -> bool { true }

    fn validate(&self) -> Result<(), String> {
        if self.name.trim().is_empty() {
            return Err("Name cannot be empty".to_string());
        }
        
        if !self.email.contains('@') {
            return Err("Invalid email format".to_string());
        }
        
        Ok(())
    }
}

Service Composition

Advanced DI with events & metrics

pub struct ApplicationService {
    container: Arc<Container>,
    cache: Arc<dyn CacheService>,
    event_bus: Arc<dyn EventBus>,
    metrics: Arc<dyn MetricsCollector>,
}

impl ApplicationService {
    pub async fn initialize(&self) -> Result<()> {
        // Set up event handlers with dependency injection
        let user_handler = Arc::new(
            UserActivityHandler::new(
                self.cache.clone(),
                self.metrics.clone(),
            )
        );
        
        let _subscription = self.event_bus
            .subscribe(user_handler)
            .await?;
            
        self.metrics.set_gauge(
            "application.startup_time", 
            1.5, 
            None
        );
        
        Ok(())
    }
}

V2 Middleware Pattern

Clean middleware with handle(request, next)

use elif_http::{Middleware, ElifRequest, ElifResponse, Next};
use elif_core::Container;
use std::sync::Arc;

pub struct AuthMiddleware {
    container: Arc<Container>,
}

impl Middleware for AuthMiddleware {
    async fn handle(
        &self,
        request: ElifRequest,
        next: Next,
    ) -> Result<ElifResponse, HttpError> {
        // Extract and validate token
        let token = request
            .header("Authorization")
            .and_then(|h| h.strip_prefix("Bearer "))
            .ok_or_else(|| HttpError::unauthorized("Missing token"))?;

        // Verify with injected service
        let auth_service = self.container.resolve::<AuthService>().await?;
        let user = auth_service.verify_token(token).await?;
        
        // Add user to request context
        request.set_context(user);
        
        // Continue to next middleware/handler
        next.run(request).await
    }
}

Quick Start

Complete application with DI and middleware

use elif_core::{Container, ServiceProvider};
use elif_http::{Server, HttpConfig, ElifRouter};
use elif_orm::DatabaseProvider;
use elif_security::{CorsMiddleware, RateLimitMiddleware};

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    // Set up dependency injection
    let container = Container::new()
        .provider(DatabaseProvider::from_env())
        .module(UserModule)
        .module(AuthModule);

    // Configure routes with middleware
    let router = ElifRouter::new()
        .middleware(CorsMiddleware::permissive())
        .middleware(RateLimitMiddleware::new(100))
        .group("/api", |api| {
            api.resource("/users", UserController)
               .resource("/posts", PostController)
        });
    
    // Start server with graceful shutdown
    Server::with_container(container, HttpConfig::from_env())?  
        .use_router(router)
        .listen("0.0.0.0:3000")
        .await
}
