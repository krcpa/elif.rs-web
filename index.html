use elif::ws::{WebSocket, Message};

async fn websocket_handler(ws: WebSocket) {
    let (tx, rx) = ws.split();
    
    let broadcast = rx
        .filter_map(|msg| msg.ok())
        .filter_map(|msg| msg.to_text().ok())
        .map(|text| Message::text(format!("Echo: {}", text)));
    
    broadcast.forward(tx).await;
}
