Quick Start

Install, create, run

1

Install CLI

cargo install elifrs
2

Create App

elifrs new my-app && cd my-app
3

Run

cargo run

Minimal server

use elif_http::*;

async fn hello() -> Response {
    Response::text("Hello")
}

let server = HttpServer::new(container)
    .route_get("/", hello)
    .bind("0.0.0.0:3000")
    .await?;

server.serve().await?;
