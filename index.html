Build a REST API

use elif_core::*;
use elif_orm::*;

#[derive(Model, Serialize, Deserialize)]
struct User {
    id: Option<Uuid>,
    email: String,
    name: String,
}

async fn create_user(req: Request) -> Response {
    let user_data: CreateUserRequest = req.json()?;
    let user = User::create(req.db(), user_data).await?;
    Response::json(user).status(201)
}

fn main() {
    let app = Application::builder()
        .provider(DatabaseProvider)
        .route("POST /users", create_user)
        .build()?;
        
    app.start().await?;
}

UserController: active users

pub async fn active_users(
    &self,
    State(container): State<Arc<Container>>,
    Query(params): Query<PaginationParams>,
) -> HttpResult<Response> {
    let pool = self.get_pool(&container).await?;
    let users = User::query()
        .select("*")
        .where_clause("is_active = $1")
        .order_by("created_at DESC")
        .limit(params.per_page.unwrap_or(20) as i32)
        .execute(&pool)
        .await
        .map_err(|e| HttpError::database_error(&e.to_string()))?;

    let api_response = ApiResponse::success(users);
    Ok(( StatusCode::OK, Json(api_response) ).into_response())
}
