Clean Middleware Pattern

The familiar handle(request, next) pattern you already know

use elif_http::{Middleware, Request, Response, Next};

#[derive(Clone)]
struct RateLimitMiddleware {
    requests_per_minute: u32,
}

#[async_trait]
impl Middleware for RateLimitMiddleware {
    async fn handle(&self, req: Request, next: Next) -> Response {
        // Check rate limit from cache
        let key = format!("rate_limit:{}", req.ip());
        let count = cache::get::<u32>(&key).await.unwrap_or(0);
        
        if count >= self.requests_per_minute {
            return Response::too_many_requests();
        }
        
        // Increment counter
        cache::increment(&key, 60).await;
        
        // Continue to next middleware
        next.run(req).await
    }
}

Elegant ORM Queries

Type-safe database operations that feel natural

use elif_orm::prelude::*;

// Define your model
#[derive(Model, Serialize, Deserialize)]
struct Article {
    id: Uuid,
    title: String,
    content: Text,
    author_id: Uuid,
    published: bool,
    #[timestamps]
    created_at: DateTime,
    updated_at: DateTime,
}

// Query with relationships
let articles = Article::query()
    .with("author")
    .with("comments.user")
    .where_("published", true)
    .order_by("created_at", Desc)
    .limit(10)
    .await?;

// Transactions made simple
db::transaction(async {
    let article = Article::create(new_article).await?;
    ActivityLog::record("article.created", &article).await?;
    Ok(article)
}).await?;

Real-time WebSockets

Channel-based WebSocket handling that scales

use elif_websocket::{Channel, Message};

#[derive(Channel)]
struct ChatChannel;

impl ChatChannel {
    async fn join(&self, socket: &WebSocket, params: Params) {
        let room_id = params.get("room_id")?;
        let user = socket.get_user()?;
        
        // Join the room
        socket.subscribe(format!("chat:{}", room_id)).await;
        
        // Notify others
        self.broadcast(
            &format!("chat:{}", room_id),
            Message::new("user_joined")
                .data("user", &user)
        ).await;
    }
    
    async fn handle_message(&self, socket: &WebSocket, msg: Message) {
        match msg.event.as_str() {
            "message" => self.handle_chat_message(socket, msg).await,
            "typing" => self.handle_typing(socket, msg).await,
            _ => {}
        }
    }
}

Dependency Injection

Wire complex applications with zero boilerplate

use elif_core::{Container, Injectable, Service};

// Define your services
#[derive(Injectable)]
struct EmailService {
    mailer: Arc<Mailer>,
    template_engine: Arc<TemplateEngine>,
}

#[derive(Injectable)]
struct NotificationService {
    email: Arc<EmailService>,
    push: Arc<PushService>,
    sms: Arc<SmsService>,
}

// Everything wires automatically
let container = Container::new()
    .singleton::<Mailer>()
    .singleton::<TemplateEngine>()
    .singleton::<EmailService>()
    .singleton::<PushService>()
    .singleton::<SmsService>()
    .singleton::<NotificationService>();

// Use anywhere
let notifications = container.resolve::<NotificationService>().await;
notifications.send_welcome(&user).await?;
