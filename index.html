Smart Controllers

Advanced search with validation & caching

impl Controller<User> for UserController {
    async fn search(
        &self,
        container: Arc<Container>,
        query: SearchParams,
    ) -> HttpResult<ElifResponse> {
        let pool = self.get_pool(&container).await?;
        
        let search_term = query.q.unwrap_or_default();
        if search_term.len() < 3 {
            return Err(HttpError::bad_request(
                "Search term must be at least 3 characters"
            ));
        }

        let users = User::query()
            .where_clause("name ILIKE $1 OR email ILIKE $1")
            .limit(50)
            .execute_with_params(&pool, vec![
                format!("%{}%", search_term)
            ])
            .await?;

        Ok(ElifResponse::json(ApiResponse::success(users)))
    }
}

Type-Safe ORM

Models with validation & timestamps

#[derive(Debug, Serialize, Deserialize, Clone)]
pub struct User {
    pub id: Uuid,
    pub name: String,
    pub email: String,
    pub age: Option<i32>,
    pub is_active: bool,
    pub created_at: DateTime<Utc>,
    pub updated_at: DateTime<Utc>,
}

impl Model for User {
    type PrimaryKey = Uuid;

    fn table_name() -> &'static str { "users" }
    fn uses_timestamps() -> bool { true }

    fn validate(&self) -> Result<(), String> {
        if self.name.trim().is_empty() {
            return Err("Name cannot be empty".to_string());
        }
        
        if !self.email.contains('@') {
            return Err("Invalid email format".to_string());
        }
        
        Ok(())
    }
}

Service Composition

Advanced DI with events & metrics

pub struct ApplicationService {
    container: Arc<Container>,
    cache: Arc<dyn CacheService>,
    event_bus: Arc<dyn EventBus>,
    metrics: Arc<dyn MetricsCollector>,
}

impl ApplicationService {
    pub async fn initialize(&self) -> Result<()> {
        // Set up event handlers with dependency injection
        let user_handler = Arc::new(
            UserActivityHandler::new(
                self.cache.clone(),
                self.metrics.clone(),
            )
        );
        
        let _subscription = self.event_bus
            .subscribe(user_handler)
            .await?;
            
        self.metrics.set_gauge(
            "application.startup_time", 
            1.5, 
            None
        );
        
        Ok(())
    }
}

Security Middleware

CORS with strict security settings

impl CorsMiddleware {
    /// Create CORS middleware with strict security
    pub fn strict() -> Self {
        let mut allowed_origins = HashSet::new();
        allowed_origins.insert(
            "https://localhost:3000".to_string()
        );
        
        Self::new(CorsConfig {
            allowed_origins: Some(allowed_origins),
            allow_credentials: true,
            max_age: Some(300), // 5 minutes
            ..CorsConfig::default()
        })
    }
}

impl Default for JwtMiddlewareConfig {
    fn default() -> Self {
        Self {
            header_name: "Authorization".to_string(),
            token_prefix: "Bearer ".to_string(),
            skip_paths: vec![
                "/health".to_string(), 
                "/metrics".to_string()
            ],
            optional: false,
        }
    }
}

Quick Start

Get running in 60 seconds

use elif_core::Application;
use elif_http::Router;

#[tokio::main]
async fn main() -> Result<()> {
    let app = Application::builder()
        .provider(DatabaseProvider)
        .provider(CacheProvider)
        .module(UserModule)
        .build().await?;

    let router = Router::new()
        .get("/api/users", list_users)
        .post("/api/users", create_user)
        .middleware(CorsMiddleware::strict())
        .middleware(JwtMiddleware::default());

    app.serve(router).await
}
