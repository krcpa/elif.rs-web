use elif_core::{Container, Application};
use elif_http::{Server, Router, Response, HttpResult};

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    // Build application with DI container
    let app = Application::builder()
        .config_file("config/app.toml")?
        .provider(DatabaseProvider)
        .provider(CacheProvider)
        .module(UserModule)
        .build().await?;

    // Define routes with framework-native types
    let router = Router::new()
        .get("/health", health_check)
        .group("/api/v1", |r| {
            r.get("/users", list_users)
             .post("/users", create_user)
             .get("/users/:id", get_user)
        });

    // Start server with graceful shutdown
    app.serve(router).await
}

async fn health_check() -> HttpResult<Response> {
    Ok(Response::json(json!({
        "status": "healthy",
        "version": env!("CARGO_PKG_VERSION")
    })))
}
use elif_http::{Controller, HttpResult};
use elif_orm::{Model, QueryBuilder};

#[derive(Model, Serialize, Deserialize)]
#[table_name = "users"]
pub struct User {
    #[primary_key]
    pub id: Uuid,
    pub email: String,
    pub name: String,
    #[serde(skip_serializing)]
    pub password_hash: String,
    pub created_at: DateTime<Utc>,
}

#[derive(Controller)]
#[model(User)]
#[routes(prefix = "/api/users")]
pub struct UserController;

impl UserController {
    // GET /api/users
    pub async fn index(
        &self,
        query: Query<PaginationParams>,
    ) -> HttpResult<Json<Page<User>>> {
        let users = User::query()
            .select(&["id", "email", "name", "created_at"])
            .order_by("created_at DESC")
            .paginate(query.page, query.per_page)
            .await?;
        
        Ok(Json(users))
    }

    // POST /api/users
    pub async fn create(
        &self,
        Json(input): Json<CreateUserDto>,
    ) -> HttpResult<Created<Json<User>>> {
        let user = User {
            id: Uuid::new_v4(),
            email: input.email,
            name: input.name,
            password_hash: hash_password(&input.password)?,
            created_at: Utc::now(),
        };
        
        let saved = user.save().await?;
        Ok(Created(Json(saved)))
    }

    // Custom endpoint: GET /api/users/search
    #[route(GET, "/search")]
    pub async fn search(
        &self,
        query: Query<SearchParams>,
    ) -> HttpResult<Json<Vec<User>>> {
        let users = User::query()
            .where_like("name", format!("%{}%", query.q))
            .or_where_like("email", format!("%{}%", query.q))
            .limit(20)
            .await?;
        
        Ok(Json(users))
    }
}
use elif_http::{Middleware, Request, Response, Next};
use elif_security::*;

// Configure comprehensive security middleware
let security = SecurityMiddleware::builder()
    .cors(Cors::default()
        .allow_origins(&["https://app.example.com"])
        .allow_methods(&[Method::GET, Method::POST])
        .allow_credentials(true))
    .csrf(Csrf::default()
        .cookie_name("csrf_token")
        .header_name("X-CSRF-Token"))
    .rate_limit(RateLimit::default()
        .max_requests(100)
        .window_secs(60))
    .headers(SecurityHeaders::default())
    .build();

// Custom authentication middleware
pub struct AuthMiddleware {
    jwt_secret: String,
}

impl Middleware for AuthMiddleware {
    async fn handle(
        &self,
        mut req: Request,
        next: Next,
    ) -> Result<Response, Error> {
        // Extract and verify JWT token
        let token = req.headers()
            .get("Authorization")
            .and_then(|h| h.to_str().ok())
            .and_then(|h| h.strip_prefix("Bearer "));

        match token {
            Some(token) => {
                let claims = verify_jwt(token, &self.jwt_secret)?;
                req.extensions_mut().insert(claims);
                next.run(req).await
            }
            None => Err(Error::unauthorized("Missing token")),
        }
    }
}

// Apply middleware to routes
let app = Router::new()
    .middleware(security)
    .middleware(Logger::default())
    .middleware(RequestId::default())
    .group("/api", |api| {
        api.middleware(AuthMiddleware { jwt_secret })
            .mount(UserController)
            .mount(PostController)
    });
use elif_orm::{Database, Transaction, QueryBuilder};
use elif_core::{Provider, Container};

// Database provider with connection pooling
pub struct DatabaseProvider;

impl Provider for DatabaseProvider {
    async fn register(&self, container: &mut Container) -> Result<()> {
        let config = container.get::<Config>()?;
        
        let pool = Database::builder()
            .url(&config.database_url)
            .max_connections(32)
            .min_connections(5)
            .connect_timeout(Duration::from_secs(30))
            .idle_timeout(Duration::from_secs(600))
            .build()
            .await?;
        
        container.singleton(pool);
        Ok(())
    }
}

// Transaction example with automatic rollback
pub async fn transfer_funds(
    db: &Database,
    from_id: Uuid,
    to_id: Uuid,
    amount: Decimal,
) -> Result<()> {
    let mut tx = db.begin().await?;
    
    // Debit from account
    Account::query()
        .where_eq("id", from_id)
        .update(json!({
            "balance": Raw::new("balance - $1", vec![amount])
        }))
        .execute(&mut tx)
        .await?;
    
    // Credit to account
    Account::query()
        .where_eq("id", to_id)
        .update(json!({
            "balance": Raw::new("balance + $1", vec![amount])
        }))
        .execute(&mut tx)
        .await?;
    
    // Create transaction record
    Transaction::create(json!({
        "from_account_id": from_id,
        "to_account_id": to_id,
        "amount": amount,
        "created_at": Utc::now(),
    }))
    .execute(&mut tx)
    .await?;
    
    tx.commit().await?;
    Ok(())
}
